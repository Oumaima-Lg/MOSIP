# 📋 MOSIP REG-MDM-108 Fix — Complete Solution Guide

## Problem Summary
```
REG-MDM-108 --> Certificate path trust validation failed
Total Number of ROOT Trust Found: 0
Total Number of INTERMEDIATE Trust Found: 0
```

**Root Cause**: CA certificates uploaded via Keymanager API go to `mosip_keymgr.keymgr.ca_cert_store`, but syncdata service reads from `mosip_master.master.ca_cert_store`. Registration client never receives certs during sync.

---

## Step 1: Generate Device Certificates with ROOT CA (No Intermediate)

### Prerequisites
- Java keytool available
- Root CA keystore: `mosiprootcaftm.p12` (password: `mosipftm`)
- Device keystores in: `Biometric Devices/{modality}/Keys/Device.p12`

### Generate New Device Certificates

```powershell
# On Windows Machine

$keytoolPath = "C:\Users\olaghjibi\MOSIP\reg-client\jre\bin\keytool.exe"
$basePath = "C:\Users\olaghjibi\MOSIP\mock-mds\collab-mock-mds-reg\target\Biometric Devices"

$devices = @(
    @{ name = "Iris Double"; path = "Iris\Double"; modality = "IRIS_DOUBLE" },
    @{ name = "Iris Single"; path = "Iris\Single"; modality = "IRIS_SINGLE" },
    @{ name = "Finger Slap"; path = "Finger\Slap"; modality = "FINGER_SLAP" },
    @{ name = "Finger Single"; path = "Finger\Single"; modality = "FINGER_SINGLE" },
    @{ name = "Face"; path = "Face"; modality = "FACE" }
)

foreach ($device in $devices) {
    $devicePath = "$basePath\$($device.path)\Keys"
    Write-Host "Processing: $($device.name)"
    
    # Delete old device cert
    Remove-Item "$devicePath\Device.p12" -Force -ErrorAction SilentlyContinue
    
    # Generate new device key with EXPLICIT 3650 days validity
    & $keytoolPath -genkeypair `
      -alias Device `
      -keyalg RSA -keysize 2048 `
      -validity 3650 `
      -keystore "$devicePath\Device.p12" `
      -storetype PKCS12 `
      -storepass mosipface `
      -keypass mosipface `
      -dname "CN=Mock $($device.name) Device, OU=MOSIP, O=MOSIP, L=Bangalore, S=Karnataka, C=IN"
    
    # Create CSR
    & $keytoolPath -certreq `
      -alias Device `
      -keystore "$devicePath\Device.p12" `
      -storetype PKCS12 `
      -storepass mosipface `
      -file "$devicePath\Device.csr"
    
    # Sign DIRECTLY with ROOT CA (2-tier chain, no intermediate)
    & $keytoolPath -gencert `
      -alias mosipftm `
      -keystore "$basePath\mosiprootcaftm.p12" `
      -storetype PKCS12 `
      -storepass mosipftm `
      -infile "$devicePath\Device.csr" `
      -outfile "$devicePath\Device.cer" `
      -validity 3650 `
      -rfc
    
    # Import root CA
    & $keytoolPath -import `
      -alias mosipftm `
      -keystore "$devicePath\Device.p12" `
      -storetype PKCS12 `
      -storepass mosipface `
      -file "$basePath\rootcaftm.cer" `
      -noprompt
    
    # Import signed device cert
    & $keytoolPath -import `
      -alias Device `
      -keystore "$devicePath\Device.p12" `
      -storetype PKCS12 `
      -storepass mosipface `
      -file "$devicePath\Device.cer" `
      -noprompt
    
    Write-Host "✓ $($device.name) updated"
}

Write-Host "`n✅ All devices updated with ROOT CA signatures"
```

## Step 3: ⚠️ **CRITICAL FIX** — Copy Certs from keymgr to master DB

This is the **KEY STEP** that resolves REG-MDM-108!

### Verify the Problem

```bash

docker exec -i postgres psql -U postgres -d mosip_keymgr -c \
  "SELECT cert_id, partner_domain FROM keymgr.ca_cert_store WHERE partner_domain='DEVICE';"

# Certs in master (will be EMPTY — this is the bug!)

  "SELECT cert_id, partner_domain FROM master.ca_cert_store WHERE partner_domain='DEVICE';"
```

### Copy Certificates Using Python Script

```bash
# On Linux Server
python3 << 'EOF'
import subprocess
import json

print("🔄 Fetching certificates from mosip_keymgr...")

# Get certs from keymgr
dump = subprocess.run(
    ['docker', 'exec', '-i', 'postgres', 'psql', '-U', 'postgres', '-d', 'mosip_keymgr',
     '-t', '-c',
     """SELECT row_to_json(t) FROM (
          SELECT cert_id, cert_subject, cert_issuer, issuer_id,
                 cert_not_before, cert_not_after,
                 COALESCE(crl_uri,'') as crl_uri, cert_data, 
                 COALESCE(cert_thumbprint,'') as cert_thumbprint,
                 COALESCE(cert_serial_no,'') as cert_serial_no, 
                 partner_domain,
                 COALESCE(cr_by,'system') as cr_by, cr_dtimes
          FROM keymgr.ca_cert_store 
          WHERE partner_domain='DEVICE'
          ORDER BY cr_dtimes DESC
        ) t"""],
    capture_output=True, text=True
)

rows = []
for line in dump.stdout.strip().split('\n'):
    line = line.strip()
    if line:
        try:
            rows.append(json.loads(line))
        except:
            pass

print(f"✅ Found {len(rows)} DEVICE certificates")

if len(rows) == 0:
    print("❌ No DEVICE certificates found in keymgr!")
    exit(1)

# Insert into master
for r in rows:
    cert_data_escaped = r['cert_data'].replace("'", "''")
    sql = f"""INSERT INTO master.ca_cert_store (
  cert_id, cert_subject, cert_issuer, issuer_id,
  cert_not_before, cert_not_after, crl_uri, cert_data,
  cert_thumbprint, cert_serial_no, partner_domain,
  cr_by, cr_dtimes, is_deleted
) VALUES (
  '{r['cert_id']}', '{r['cert_subject']}', '{r['cert_issuer']}', '{r['issuer_id']}',
  '{r['cert_not_before']}', '{r['cert_not_after']}', '{r['crl_uri']}', '{cert_data_escaped}',
  '{r['cert_thumbprint']}', '{r['cert_serial_no']}', '{r['partner_domain']}',
  '{r['cr_by']}', '{r['cr_dtimes']}', false
) ON CONFLICT (cert_id) DO NOTHING;"""

    result = subprocess.run(
        ['docker', 'exec', '-i', 'postgres', 'psql', '-U', 'masteruser', '-d', 'mosip_master', '-c', sql],
        capture_output=True, text=True
    )
    if 'ERROR' in result.stderr or 'ERROR' in result.stdout:
        print(f"❌ FAILED {r['cert_id']}: {result.stderr}")
    else:
        print(f"✅ OK: {r['cert_id']}")

print("✅ All certificates copied!")
EOF
```
